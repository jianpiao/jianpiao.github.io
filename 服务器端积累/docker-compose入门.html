# 初始化项目

npm init -y

修改一下脚本，给个项目启动命令，在package.json

{
  "scripts": {
    "start": "nodemon index.js"
  },
}

新增一些必须要的依赖包

yarn add koa koa-router nodemon

在根目录下创建一个index.js文件写入如下内容

//  server.js
const Koa = require('koa')
const Router = require('koa-router')
const mysql = require('promise-mysql')

let app = new Koa()
let router = new Router()

// 根路由
router.get('/', async ctx => {
  ctx.body = 'index'
})

router.get('/get', async ctx => {
  ctx.body = '获取内容'
})

router.get('/db', async ctx => {
  let db = await mysql.createPool({
    host: 'mysql', // 这里注意要和mysql镜像同名
    port: 3306,
    user: 'root',
    password: '123456'
  })
  let res = await db.query('SHOW DATABASES')
  ctx.body = res
})

app.use(router.routes())
app.listen(3000)

# docker-compose

这一步就是开始进行docker-compose配置。 compose项目是Docker官方的开源项目，负责实现对Docker容器集群的快速编排。 Compose 定位是 「定义和运行多个 Docker 容器的应用（Defining and running multi-container Docker applications）」，其前身是开源项目 Fig。

Compose有两个重要的概念：

  • 服务 (service)：一个应用的容器，实际上可以包括若干运行相同镜像的容器实例。
  • 项目 (project)：由一组关联的应用容器组成的一个完整业务单元，在 docker-compose.yml 文件中定义。

Compose的默认管理对象是项目，通过子命令对项目的一组容器进行便捷的生命周期管理。 概念了解之后，在跟目录下新建一个docker-compose.yml文件，由于我们要集成三个项目，所以配置三个项目。

version: "3.1"
services:
  web:
    image: node:14-alpine #node镜像名称
    working_dir: /code #工作目录
    volumes:
      #挂载点
      - .:/code #把本地当前目录挂载到容器的/code目录
    ports:
      #端口映射
      - "3000:3000" #本地端口：容器端口
    command: yarn start #这条命令会在工作目录下执行

  nginx:
    image: nginx:1.17
    volumes:
      - ./static:/code/static
      - ./conf/default.conf:/etc/nginx/conf.d/default.conf
    ports:
      - "80:80"

  mysql:
    image: mysql:8.0.28-oracle
    command: --default-authentication-plugin=mysql_native_password
    restart: always
    volumes:
      - ./data:/var/lib/mysql #挂载当前的data目录到容器默认mysql存储目录
    environment:
      - MYSQL_ROOT_PASSWORD=123456
      - MYSQL_DATABASE=test
    ports:
      - "3306:3306"

compose有很多命令，具体可以参考地址

# Nginx

由于需要修改反向代理的配置文件，所以在根目录下新建一个文件夹：conf，并新建一个文件default.conf 在文件内容写入：

server {
    listen       80;
    server_name  localhost;

    location /static {
        alias /code/static;             #路径可以自定义，放在什么目录都可以
    }

    location / {
        proxy_pass http://web:3000;     #地址web是docker-compose.yml里面的web服务名
    }
}

新建完成之后，执行命令

docker-compose up

会按照配置的内容拉取镜像并运行容器。

# 总结

以上就是一个正常的流程，其实很简单。 如果在浏览器中访问localhost可以不需要写端口号，假入出问题，可以去检查一下nginx里面的

/etc/nginx/conf.d/default.conf

是否有被修改，如果没被修改，则先查询docker psnginx的容器id，然后根据容器id复制

# 复制nginx容器里的配置文件到当前目录
docker cp 35c5c3671780:/etc/nginx/conf.d/default.conf ./conf/

最后重新执行一遍docker任务

docker-compose up

# 或者
# 删除之前的镜像，重新生成新的镜像和容器
docker-compose up -d --build
